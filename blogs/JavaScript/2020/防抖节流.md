## 防抖

事件响应函数在一段时间以后才执行
如果这段时间内再次调用，则重新计算执行时间；
当预定时间内没有再次调用，则执行函数。

应用场景
1.scroll时间滚动触发
2.搜索框输入查询
2.表单验证
4。按钮提交事件
5.浏览器窗口缩放，resize事件

```javascript
    //防抖函数
    function debounce(func,wait,immediate){
        let timeout,result;
        let debounced=function(){
            //改变执行函数this指向问题
            let context=this;
            let args=arguments;
            clearTimeout(timeout);
            if(immediate){
                let callNow=!timeout;
                timeout=setTimeout(()=>{
                    timeout=null
                },wait)
                    //立即执行
                if(callNow){
                window.console.log(timeout)
                func.apply(context,args)
                } 
            }else{
                //延迟执行 1秒后执行
                timeout=setTimeout(function(){
                //改变执行函数this,e指向问题
                func.apply(context,args)
                },wait)
            } 
            return result
        }
        // 取消操作
        debounced.cancel=function(){
            clearTimeout(timeout)
            //解决内存泄漏问题
            timeout=null
        }
        return debounced
    }
    //演示事件是如何频繁发生
    let count=0;
    let container =document.querySelector('.demo')
    let btn=document.querySelector('#btn')
    //执行函数
    function doSomething(e){
        //event指向问题 应该指向当前debounce防抖函数
        console.log(e)

        console.log(this)//应该指向当前容器 container
        container.innerHTML=count++
        //改变执行函数内部this的指向
    }

    const doSome=debounce(doSomething,1000,true)
    //取消操作
    btn.onclick=function(){
        doSome.cancel
    }
    container.onmousemove= doSome
```

## 节流

如果持续触发事件，每隔一段时间，只会执行一次事件
